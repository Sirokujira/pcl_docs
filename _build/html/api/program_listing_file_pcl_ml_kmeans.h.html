      void
      computeCentroids();

      // distance between two points
      float distance(const Point& x, const Point& y)
      {
        float total = 0.0;
        float diff;

        auto cpy=y.cbegin();
        for(auto cpx = x.cbegin(), cpx_end = x.cend(); cpx != cpx_end; ++cpx, ++cpy){
          diff = *cpx - *cpy;
          total += (diff * diff);
        }
        return total;  // no need to take sqrt, which is monotonic
      }
